## Overview

The BitTorrent community has built various tools to integrate TrackersList with popular clients and workflows. These tools automatically fetch and add trackers to your torrents or client configurations.

<Info>
  These are **third-party projects** maintained by their respective authors, not by the TrackersList maintainer.
</Info>

## Building Your Own Tool

### Quick Integration Guide

Want to build your own TrackersList integration? Here's how:

<Steps>
  <Step title="Choose a Mirror">
    Select the most appropriate mirror for your use case:
    
    ```javascript
    const MIRRORS = [
      'https://cdn.jsdelivr.net/gh/ngosang/trackerslist@master/trackers_best.txt',
      'https://ngosang.github.io/trackerslist/trackers_best.txt',
      'https://raw.githubusercontent.com/ngosang/trackerslist/master/trackers_best.txt'
    ];
    ```
  </Step>
  
  <Step title="Fetch the List">
    Download and parse the tracker list:
    
    ```python Python Example
    import requests
    
    def fetch_trackers():
        url = 'https://cdn.jsdelivr.net/gh/ngosang/trackerslist@master/trackers_best.txt'
        response = requests.get(url)
        response.raise_for_status()
        
        # Split by blank lines
        trackers = [line.strip() for line in response.text.split('\n') if line.strip()]
        return trackers
    ```
  </Step>
  
  <Step title="Integrate with Your Client">
    Add trackers using your client's API or configuration format
  </Step>
  
  <Step title="Implement Caching">
    Cache locally and refresh daily to respect rate limits
  </Step>
  
  <Step title="Handle Errors">
    Implement failover and graceful degradation
  </Step>
</Steps>

```python Python Client Integration
import requests
import json
from datetime import datetime, timedelta

class TrackersListClient:
    def __init__(self, cache_file='trackers_cache.json'):
        self.cache_file = cache_file
        self.mirrors = [
            'https://cdn.jsdelivr.net/gh/ngosang/trackerslist@master/trackers_best.txt',
            'https://ngosang.github.io/trackerslist/trackers_best.txt',
            'https://raw.githubusercontent.com/ngosang/trackerslist/master/trackers_best.txt'
        ]
    
    def get_trackers(self, list_name='trackers_best', force_refresh=False):
        # Check cache
        if not force_refresh:
            cached = self._load_cache()
            if cached:
                return cached
        
        # Fetch from mirrors with failover
        for mirror in self.mirrors:
            try:
                url = mirror.replace('trackers_best.txt', f'{list_name}.txt')
                response = requests.get(url, timeout=10)
                response.raise_for_status()
                
                trackers = [line.strip() for line in response.text.split('\n') if line.strip()]
                self._save_cache(trackers)
                return trackers
            except requests.RequestException:
                continue
        
        raise Exception('All mirrors failed')
    
    def _load_cache(self):
        try:
            with open(self.cache_file, 'r') as f:
                data = json.load(f)
                timestamp = datetime.fromisoformat(data['timestamp'])
                if datetime.now() - timestamp < timedelta(hours=24):
                    return data['trackers']
        except (FileNotFoundError, KeyError, ValueError):
            pass
        return None
    
    def _save_cache(self, trackers):
        with open(self.cache_file, 'w') as f:
            json.dump({
                'timestamp': datetime.now().isoformat(),
                'trackers': trackers
            }, f)

# Usage
client = TrackersListClient()
trackers = client.get_trackers()
print(f'Fetched {len(trackers)} trackers')
```

```javascript Node.js Client Integration
const fs = require('fs').promises;
const path = require('path');

class TrackersListClient {
  constructor(cacheFile = 'trackers_cache.json') {
    this.cacheFile = cacheFile;
    this.mirrors = [
      'https://cdn.jsdelivr.net/gh/ngosang/trackerslist@master/trackers_best.txt',
      'https://ngosang.github.io/trackerslist/trackers_best.txt',
      'https://raw.githubusercontent.com/ngosang/trackerslist/master/trackers_best.txt'
    ];
  }
  
  async getTrackers(listName = 'trackers_best', forceRefresh = false) {
    // Check cache
    if (!forceRefresh) {
      const cached = await this._loadCache();
      if (cached) return cached;
    }
    
    // Fetch from mirrors with failover
    for (const mirror of this.mirrors) {
      try {
        const url = mirror.replace('trackers_best.txt', `${listName}.txt`);
        const response = await fetch(url, { signal: AbortSignal.timeout(10000) });
        
        if (!response.ok) continue;
        
        const text = await response.text();
        const trackers = text.split('\n').filter(line => line.trim());
        
        await this._saveCache(trackers);
        return trackers;
      } catch (error) {
        continue;
      }
    }
    
    throw new Error('All mirrors failed');
  }
  
  async _loadCache() {
    try {
      const data = JSON.parse(await fs.readFile(this.cacheFile, 'utf8'));
      const timestamp = new Date(data.timestamp);
      const age = Date.now() - timestamp.getTime();
      
      if (age < 24 * 60 * 60 * 1000) {
        return data.trackers;
      }
    } catch (error) {
      // Cache doesn't exist or is invalid
    }
    return null;
  }
  
  async _saveCache(trackers) {
    await fs.writeFile(this.cacheFile, JSON.stringify({
      timestamp: new Date().toISOString(),
      trackers
    }));
  }
}

// Usage
const client = new TrackersListClient();
client.getTrackers().then(trackers => {
  console.log(`Fetched ${trackers.length} trackers`);
});
```

```bash Bash Integration
#!/bin/bash

# Simple bash function to fetch trackers
fetch_trackers() {
    local list_name="${1:-trackers_best}"
    local cache_file="/tmp/trackerslist_${list_name}.txt"
    local cache_age_hours=24
    
    # Check if cache exists and is fresh
    if [ -f "$cache_file" ]; then
        local cache_age=$(($(date +%s) - $(stat -f %m "$cache_file" 2>/dev/null || stat -c %Y "$cache_file")))
        if [ $cache_age -lt $((cache_age_hours * 3600)) ]; then
            cat "$cache_file"
            return 0
        fi
    fi
    
    # Try mirrors with failover
    local mirrors=(
        "https://cdn.jsdelivr.net/gh/ngosang/trackerslist@master/${list_name}.txt"
        "https://ngosang.github.io/trackerslist/${list_name}.txt"
        "https://raw.githubusercontent.com/ngosang/trackerslist/master/${list_name}.txt"
    )
    
    for mirror in "${mirrors[@]}"; do
        if curl -sf --max-time 10 "$mirror" -o "$cache_file"; then
            cat "$cache_file"
            return 0
        fi
    done
    
    echo "Error: All mirrors failed" >&2
    return 1
}

# Usage
fetch_trackers "trackers_best" | grep -v '^$'
```
